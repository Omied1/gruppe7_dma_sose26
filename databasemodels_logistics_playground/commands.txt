## docker

# docker compose ausführen
docker compose up -d

# container stoppen und entfernen, images aber nicht entfernen
docker compose down (optional -v) für volumes löschen
#container nur stoppen (alternativ: starten)
docker compose stop (alternativ: start)


# container anzeigen
docker ps

# in container springen
docker exec -it 1b429959a53b /bin/bash

# alles von Rechner löschen
docker compose down -v --rmi all --remove-orphans
